 Q1] #include
        int main()
        {
           int i,n,count=0;
           printf("Enter the number:");
           scanf("%d",&n);
           
           for(i=1;i<=n;i++)
           {
              if(n%i==0)
              count++;
           }
           if (count==2)
           printf("It is Prime Number.");
           else
           printf("%d is not a Prime Number.",n);
        
          return 0;
        
        
        }
